ngSecuredProvider.configAuth = {

    pages: {
        login: "app.login",
        unAuthorized: "app.unauth"

    },
    authenticating: function($http, ngSecured){

        return $http.get("/loginUrl", ngSecured.credentials).then(function(data){
            ngSecured.setToken(data.token);
            ngSecured.setPermissions(data.permissions);
            ngSecured.setRoles(data.roles);
        });

    },


    loginUrl: "/login",
    authorizing: function(){

    }
}
